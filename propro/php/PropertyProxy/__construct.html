#void php\PropertyProxy::__construct(mixed &$object, string $member[, php\PropertyProxy $parent = NLL])

Create a new property proxy for the property $member of $object.

Params:#

  • mixed reference &$object# The storage from which to reference the property.
  • string $member# The name of the property to reference.
  • php\PropertyProxy $parent# = NULL Any parent property proxy.

Example:# 


class c {
    private $storage = array();
    function __get($p) {
        return new php\PropertyProxy($this->storage, $p);
    }
    function __set($p, $v) {
        $this->storage[$p] = $v;
    }
}

$c = new c;
$c->data["foo"] = 1;

var_dump($c);

$c->data[] = 1;
$c->data[] = 2;
$c->data[] = 3;
$c->data["bar"][] = 123;
$c->data["bar"][] = 456;

var_dump($c);
unset($c->data["bar"][0]);

var_dump($c);
